function hasArtifactStorage() {
  return typeof window !== "undefined" && window.storage && typeof window.storage.get === "function";
}
async function loadPersisted(key, fallback) {
  try {
    if (hasArtifactStorage()) {
      const res = await window.storage.get(key, true);
      return res ? JSON.parse(res.value) : fallback;
    }
  } catch (e) {}
  try {
    const raw = localStorage.getItem(key);
    return raw ? JSON.parse(raw) : fallback;
  } catch (e) {
    return fallback;
  }
}
async function savePersisted(key, value) {
  try {
    if (hasArtifactStorage()) {
      await window.storage.set(key, JSON.stringify(value), true);
      return;
    }
  } catch (e) {}
  try { localStorage.setItem(key, JSON.stringify(value)); } catch (e) {}
}

        code: `DTR-${label.slice(0, 2).toUpperCase()}-${String(i + 1).padStart(3, "0")}`,

        status: "Active",
      };
    });

function SimplePieChart({ data, colors, size = 200 }) {
  const total = data.reduce((s, d) => s + d.qty, 0) || 1;
  let angle = 0;
  const r = size / 2 - 6, cx = size / 2, cy = size / 2;
  const slices = data.map((d, i) => {
    const frac = d.qty / total;
    const start = angle;
    const end = angle + frac * 360;
    angle = end;
    const large = end - start > 180 ? 1 : 0;
    const toXY = (ang) => {
      const rad = (ang - 90) * Math.PI / 180;
      return [cx + r * Math.cos(rad), cy + r * Math.sin(rad)];
    };
    const [x1, y1] = toXY(start), [x2, y2] = toXY(end);
    const path = frac >= 0.999
      ? `M ${cx} ${cy - r} A ${r} ${r} 0 1 1 ${cx - 0.01} ${cy - r} Z`
      : `M ${cx} ${cy} L ${x1} ${y1} A ${r} ${r} 0 ${large} 1 ${x2} ${y2} Z`;
    return <path key={i} d={path} fill={colors[i % colors.length]} stroke="white" strokeWidth="1.5" />;
  });
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 20, flexWrap: "wrap" }}>
      <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>{slices}</svg>
      <div>
        {data.map((d, i) => (
          <div key={i} style={{ display: "flex", alignItems: "center", gap: 8, fontSize: 12.5, marginBottom: 6 }}>
            <span style={{ width: 11, height: 11, borderRadius: 3, background: colors[i % colors.length], display: "inline-block" }} />
            {d.group}: <strong>{d.qty}</strong>
          </div>
        ))}
      </div>
    </div>
  );
}

          {tab === "transfer" && <TransferForm materials={materials} onSubmit={updateQty} showToast={showToast} />}
          {tab === "deadstock" && <DeadStock materials={materials} txns={txns} onSubmit={updateQty} showToast={showToast} />}
          {tab === "reports" && <Reports materials={materials} txns={txns} onExport={exportCSV} />}
        </main>
      </div>

      {toast && (
        <div style={{ position: "fixed", bottom: 20, right: 20, background: toast.kind === "success" ? "#166534" : "#b91c1c", color: "white", padding: "10px 16px", borderRadius: 8, fontSize: 13.5, display: "flex", alignItems: "center", gap: 8, boxShadow: "0 6px 20px rgba(0,0,0,0.2)", zIndex: 50 }}>

        </div>
      )}
    </div>
  );
}

function useMaterialSelect(materials) {
  const [materialId, setMaterialId] = useState(materials[0]?.id || "");
  useEffect(() => { if (!materials.find((m) => m.id === materialId) && materials[0]) setMaterialId(materials[0].id); }, [materials]);
  return [materialId, setMaterialId];
}

function FormShell({ title, children, onSubmit, submitLabel, submitColor }) {
  return (
    <form onSubmit={onSubmit} style={{ background: "white", borderRadius: 10, border: "1px solid #e2e8f0", padding: 20, maxWidth: 640 }}>
      <div style={{ fontWeight: 700, fontSize: 16, marginBottom: 14, color: BLUE }}>{title}</div>
      <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>{children}</div>
      <button type="submit" style={{ ...btnStyle(submitColor || BLUE), marginTop: 16 }}><IconPlus size={14} /> {submitLabel}</button>
    </form>
  );
}

function ReceiptForm({ materials, onSubmit, showToast }) {
  const [materialId, setMaterialId] = useMaterialSelect(materials);
  const [qty, setQty] = useState(1);
  const [supplier, setSupplier] = useState("");
  const [invoice, setInvoice] = useState("");
  const [remarks, setRemarks] = useState("");
  const handle = (e) => {
    e.preventDefault();
    if (!materialId || qty <= 0) return;
    onSubmit(materialId, Number(qty), "Receipt", { supplier, invoice, remarks });
    showToast(`GRN recorded: +${qty} unit(s) received.`);
    setQty(1); setSupplier(""); setInvoice(""); setRemarks("");
  };
  return (
    <FormShell title="Material Receipt — Goods Receipt Note (GRN)" onSubmit={handle} submitLabel="Post Receipt">
      <div>
        <label style={fieldLabel}>Material</label>
        <select value={materialId} onChange={(e) => setMaterialId(e.target.value)} style={inputStyle}>
          {materials.map((m) => <option key={m.id} value={m.id}>{m.code} — {m.name}</option>)}
        </select>
      </div>
      <div style={{ display: "flex", gap: 12 }}>
        <div style={{ flex: 1 }}><label style={fieldLabel}>Quantity Received</label><input type="number" min={1} value={qty} onChange={(e) => setQty(e.target.value)} style={inputStyle} /></div>
        <div style={{ flex: 1 }}><label style={fieldLabel}>Invoice Number</label><input value={invoice} onChange={(e) => setInvoice(e.target.value)} style={inputStyle} placeholder="e.g. INV/2026/0142" /></div>
      </div>
      <div><label style={fieldLabel}>Supplier / Source</label><input value={supplier} onChange={(e) => setSupplier(e.target.value)} style={inputStyle} placeholder="Supplier name or repair agency" /></div>
      <div><label style={fieldLabel}>Remarks</label><textarea value={remarks} onChange={(e) => setRemarks(e.target.value)} style={{ ...inputStyle, minHeight: 60 }} /></div>
    </FormShell>
  );
}

function IssueForm({ materials, onSubmit, showToast }) {
  const [materialId, setMaterialId] = useMaterialSelect(materials);
  const [qty, setQty] = useState(1);
  const [issuedTo, setIssuedTo] = useState("");
  const [purpose, setPurpose] = useState("");
  const selected = materials.find((m) => m.id === materialId);
  const handle = (e) => {
    e.preventDefault();
    if (!materialId || qty <= 0) return;
    if (selected && Number(qty) > Number(selected.qty)) { showToast("Issue quantity exceeds available stock — negative stock is not allowed.", "error"); return; }
    onSubmit(materialId, -Number(qty), "Issue", { issuedTo, purpose });
    showToast(`Issue voucher posted: -${qty} unit(s) issued.`);
    setQty(1); setIssuedTo(""); setPurpose("");
  };
  return (
    <FormShell title="Material Issue Voucher" onSubmit={handle} submitLabel="Post Issue" submitColor={ORANGE}>
      <div>
        <label style={fieldLabel}>Material</label>
        <select value={materialId} onChange={(e) => setMaterialId(e.target.value)} style={inputStyle}>
          {materials.map((m) => <option key={m.id} value={m.id}>{m.code} — {m.name} (Avail: {m.qty})</option>)}
        </select>
      </div>
      <div style={{ display: "flex", gap: 12 }}>
        <div style={{ flex: 1 }}><label style={fieldLabel}>Quantity to Issue</label><input type="number" min={1} value={qty} onChange={(e) => setQty(e.target.value)} style={inputStyle} /></div>
        <div style={{ flex: 1 }}><label style={fieldLabel}>Issued To (Section / Sub-Division)</label><input value={issuedTo} onChange={(e) => setIssuedTo(e.target.value)} style={inputStyle} /></div>
      </div>
      <div><label style={fieldLabel}>Purpose</label><input value={purpose} onChange={(e) => setPurpose(e.target.value)} style={inputStyle} placeholder="e.g. Replacement against burnt transformer complaint" /></div>
    </FormShell>
  );
}

function TransferForm({ materials, onSubmit, showToast }) {
  const [materialId, setMaterialId] = useMaterialSelect(materials);
  const [qty, setQty] = useState(1);
  const [fromStore, setFromStore] = useState(STORES[0]);
  const [toStore, setToStore] = useState(STORES[1]);
  const selected = materials.find((m) => m.id === materialId);
  const handle = (e) => {
    e.preventDefault();
    if (!materialId || qty <= 0 || fromStore === toStore) return;
    if (selected && Number(qty) > Number(selected.qty)) { showToast("Transfer quantity exceeds available stock.", "error"); return; }
    onSubmit(materialId, 0, "Transfer", { fromStore, toStore, movedQty: qty });
    showToast(`Transfer logged: ${qty} unit(s) from ${fromStore} to ${toStore}.`);
    setQty(1);
  };
  return (
    <FormShell title="Stock Transfer Between Stores" onSubmit={handle} submitLabel="Record Transfer">
      <div>
        <label style={fieldLabel}>Material</label>
        <select value={materialId} onChange={(e) => setMaterialId(e.target.value)} style={inputStyle}>
          {materials.map((m) => <option key={m.id} value={m.id}>{m.code} — {m.name} (Avail: {m.qty})</option>)}
        </select>
      </div>
      <div style={{ display: "flex", gap: 12 }}>
        <div style={{ flex: 1 }}><label style={fieldLabel}>From Store</label><select value={fromStore} onChange={(e) => setFromStore(e.target.value)} style={inputStyle}>{STORES.map((s) => <option key={s}>{s}</option>)}</select></div>
        <div style={{ flex: 1 }}><label style={fieldLabel}>To Store</label><select value={toStore} onChange={(e) => setToStore(e.target.value)} style={inputStyle}>{STORES.map((s) => <option key={s}>{s}</option>)}</select></div>
      </div>
      <div><label style={fieldLabel}>Quantity</label><input type="number" min={1} value={qty} onChange={(e) => setQty(e.target.value)} style={inputStyle} /></div>
    </FormShell>
  );
}

function DeadStock({ materials, txns, onSubmit, showToast }) {
  const failMaterials = materials.filter((m) => m.group !== "New/Repair");
  const [materialId, setMaterialId] = useState(failMaterials[0]?.id || "");
  const [qty, setQty] = useState(1);
  const [action, setAction] = useState("Condemnation");
  const selected = materials.find((m) => m.id === materialId);
  const handle = (e) => {
    e.preventDefault();
    if (!materialId || qty <= 0) return;
    if (selected && Number(qty) > Number(selected.qty)) { showToast("Quantity exceeds available failed-transformer stock.", "error"); return; }
    onSubmit(materialId, -Number(qty), "DeadStock", { action });
    showToast(`${action} recorded for ${qty} unit(s).`);
    setQty(1);
  };
  const disposalLog = txns.filter((t) => t.type === "DeadStock");
  return (
    <div style={{ display: "flex", gap: 16, flexWrap: "wrap" }}>
      <FormShell title="Dead Stock / Scrap Disposal" onSubmit={handle} submitLabel="Record Action" submitColor="#dc2626">
        <div>
          <label style={fieldLabel}>Failed Transformer</label>
          <select value={materialId} onChange={(e) => setMaterialId(e.target.value)} style={inputStyle}>
            {failMaterials.map((m) => <option key={m.id} value={m.id}>{m.code} — {m.name} (Avail: {m.qty})</option>)}
          </select>
        </div>
        <div style={{ display: "flex", gap: 12 }}>
          <div style={{ flex: 1 }}><label style={fieldLabel}>Quantity</label><input type="number" min={1} value={qty} onChange={(e) => setQty(e.target.value)} style={inputStyle} /></div>
          <div style={{ flex: 1 }}><label style={fieldLabel}>Action</label><select value={action} onChange={(e) => setAction(e.target.value)} style={inputStyle}>{["Inspection", "Condemnation", "Auction", "Scrap Disposal"].map((a) => <option key={a}>{a}</option>)}</select></div>
        </div>
      </FormShell>
      <div style={{ background: "white", borderRadius: 10, border: "1px solid #e2e8f0", padding: 16, flex: "1 1 340px" }}>
        <div style={{ fontWeight: 700, marginBottom: 10 }}>Disposal Register</div>
        {disposalLog.length === 0 ? <div style={{ color: "#94a3b8", fontSize: 13 }}>No disposal actions recorded yet.</div> : (
          <table style={{ width: "100%", fontSize: 12.5, borderCollapse: "collapse" }}>
            <thead><tr style={{ textAlign: "left", color: "#64748b" }}><th style={{ padding: "6px 4px" }}>Date</th><th>Material</th><th>Qty</th><th>Action</th></tr></thead>
            <tbody>
              {disposalLog.map((t) => {
                const m = materials.find((mm) => mm.id === t.materialId);
                return <tr key={t.id} style={{ borderTop: "1px solid #f1f5f9" }}><td style={{ padding: "6px 4px" }}>{t.date}</td><td>{m?.name}</td><td>{t.qty}</td><td>{t.action}</td></tr>;
              })}
            </tbody>
          </table>
        )}
      </div>
    </div>
  );
}

function Reports({ materials, txns, onExport }) {
  const receipts = txns.filter((t) => t.type === "Receipt");
  const issues = txns.filter((t) => t.type === "Issue");
  const transfers = txns.filter((t) => t.type === "Transfer");
  const Section = ({ title, rows, cols }) => (
    <div style={{ background: "white", borderRadius: 10, border: "1px solid #e2e8f0", padding: 16, marginBottom: 16 }}>
      <div style={{ fontWeight: 700, marginBottom: 10 }}>{title} ({rows.length})</div>
      {rows.length === 0 ? <div style={{ color: "#94a3b8", fontSize: 13 }}>No records yet.</div> : (
        <div style={{ overflowX: "auto" }}>
          <table style={{ width: "100%", fontSize: 12.5, borderCollapse: "collapse" }}>
            <thead><tr style={{ textAlign: "left", color: "#64748b" }}>{cols.map((c) => <th key={c} style={{ padding: "6px 4px" }}>{c}</th>)}</tr></thead>
            <tbody>
              {rows.map((t) => {
                const m = materials.find((mm) => mm.id === t.materialId);
                return (
                  <tr key={t.id} style={{ borderTop: "1px solid #f1f5f9" }}>
                    <td style={{ padding: "6px 4px" }}>{t.date}</td><td>{m?.name}</td><td>{t.qty}</td>
                    <td>{t.supplier || t.issuedTo || t.fromStore || "—"}</td>
                    <td>{t.invoice || t.purpose || (t.toStore ? `→ ${t.toStore}` : "—")}</td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
  return (
    <div>
      <div style={{ display: "flex", justifyContent: "flex-end", marginBottom: 12 }}>
        <button onClick={onExport} style={btnStyle(ORANGE)}><IconDownload size={14} /> Export Stock Register (CSV)</button>
      </div>
      <Section title="Material Receipt Register" rows={receipts} cols={["Date", "Material", "Qty", "Supplier", "Invoice"]} />
      <Section title="Material Issue Register" rows={issues} cols={["Date", "Material", "Qty", "Issued To", "Purpose"]} />
      <Section title="Stock Transfer Register" rows={transfers} cols={["Date", "Material", "Qty", "From Store", "To Store"]} />
    </div>
  );
}

  return <SIMMSAppInner onLogout={() => setAuthed(false)} />;
}
